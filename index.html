#--- this use of wordcloud2 & tm & tidytext

suppressPackageStartupMessages(library(readr))
suppressPackageStartupMessages(library(wordcloud2))
suppressPackageStartupMessages(library(tm))
suppressPackageStartupMessages(library(tidytext))

suppressPackageStartupMessages(library(htmlwidgets))
suppressPackageStartupMessages(library(webshot))


#--- Take all the text from this file itself ---
myFileText <- readr::read_file("index.Rmd")

#--- Turn it into a "Corpus" object for use with tm and then wordcloud2 ---
preDocs <- VectorSource(myFileText)
docs <- Corpus(preDocs)


#--- We then clean out unhelpful characters and words with tm before counting frequencies ---

    #-- Use tm to remove characters we don't want to count --
docs <- tm_map(docs, removePunctuation) #remove punctuation
docs <- tm_map(docs, removeNumbers) #remove numbers

    #-- Use tm to convert all to lowercase because we won't consider capitals as separating words --
docs <- tm_map(docs, tolower) #convert all characters to lower case

    #-- Go ahead and remove the standard exclusion words - "stopwords" - according to tm --
docs <- tm_map(docs, removeWords, stopwords("english")) #remove common words

    #-- And then I'll also remove other words dominating the frequency count that aren't interesting --

        #- manually remove these high-frequency words that are just keywords from the markdown & styling -
myManualExclusions <- c("textaligncenter", "dfout", "manflexchronresumetext", "manualflexboxstrengthsfinder", "nbspnbsp‚Ä¢nbspnbsp", "manualflexbox", "marginleft", "marginright", "manflexsummaryleft", "manflexsummaryright", "manflexchrontimeline", "varmyaccentblue", "liactive", "transparent", "elementblank", "count", "bbp", "onetab", "twotab", "metatag", "paddingbottom", "paddingtop", "marginbottom", "margintop", "mymanualexclusions",  "onetabxs","onetabns","clicksummary","summarymore","margintoppx","sansserif","tajawal","summaryclick","fontsize","marginbottompx","marginleftpx","stylewidth","tight","borderleftwidth","columns","fontsizepx","lineheight","small","bbpnol","marginbottom","fontweight","margintop","navtabs","normal","width","borderbottomcolor","borderbottomwidth","flex","fontfamily","part","onetab","tmmapdocs","docs","myplot","styledisplay","size")
        #- manually remove these high-frequency words that are simply not descriptive or redundant to others -
myManualExclusions <- c(myManualExclusions, "education", "university", "twotabsemitight", "div", "manually", "now", "one", "characters", "geomlineaesx", "justifycontentspaceevenly", "moore", "plan", "pre", "resume", "simple", "usa", "convert", "finally", "frequencies", "great", "helped", "intern", "keywords", "large", "last", "lines", "member", "needs", "plot", "position", "process", "rmarkdown", "set", "staff", "stylemargintoppx", "term", "text", "times", "top", "wipe", "‚Äì", "‚ñå", "can", "meta", "display", "highfrequency", "freqencies", "I","click","using","developed","designed","‚Äì","‚Äî","‚Äì","new","contractor","jan","feb","mar","apr","may","jun","jul","aug","sep","oct","nov","dec","work","ohio","including","program","summary","center","chapter","well","around","first","successfully","take","taking","year","event","insurance","nationwide","state","stored","massage","truemt","mix","present","north","brands","columbus","certified","amtaohio","end","expesite","firms","lets","many","two","words","use","time","remove","used","across","america","color","difficult","every","document","kentucky","fulltime","also","career","classroom","committee","towards","groups","several","file","made","took","output","parttime","occasional", "different", "important", "implemented", "office", "working", "therapy", "therapist", "certificate", "continuing", "hub", "apps")
        # !! - manually remove these high-frequency words that are technology keywords I'm intentionally skipping -
myManualExclusions <- c(myManualExclusions, "server","sql","data","ssis","tsql","asp","web","com","net","google","application","aspnet","xml","etl","forms")
            # ^^^ this last one is where I'm "cheating" a little to drop technology keywords from my past

    #-- And use my manual exclusion list now against the Corpus --
docs <- tm_map(docs, removeWords, myManualExclusions) #remove any additional specific words

    #-- Finally, we also remove whitespace --
docs <- tm_map(docs, stripWhitespace) #remove whitespaces



#--- Then finally we use tm to create the matrix of words and frequencies ---
dtm <- DocumentTermMatrix(docs)

#--- And then with tidy we convert the document term matrix into a dataframe and sort by frequency ---
df <- tidy(dtm)
df <- df[order(-df$count),c(2,3)]



#--- I further refine the list to a reasonable top set, today those of at least 3 ---
dfOut <- df[df$count>2,]
# ---  I drop the unprintable character I can't exclude <<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<< have to keep updating
dfOut <- dfOut[-(4),]
#--- And I drop the count of "Evaluation" because it makes the difference in word sizes too large ---
dfOut[dfOut$term == "evaluation",]$count = 12




#--- And now we can output the cloud !!! ---
wordcloud2(data = dfOut, shape = 'circle', size=.25, rotateRatio = .2, ellipticity=.5, gridSize = 7)
